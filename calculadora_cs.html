Exercício: Calculadora

Escreva um programa que exiba um menu para o usuário com as seguintes opções:

  • 1 - Adição
  • 2 - Subtração
  • 3 - Multiplicação
  • 4 - Divisão

O programa deve receber a opção selecionada pelo usuário e, em seguida, solicitar os valores necessários para a operação escolhida. Após realizar o cálculo, o resultado deve ser exibido no console.

Métodos

  • Console.WriteLine(): Utilizado para exibir mensagens no console.
  • Console.ReadLine(): Utilizado para ler a entrada do usuário.
  • int.TryParse(): Utilizado para converter a entrada do usuário em um número inteiro.
  • switch-case: Utilizado para executar diferentes blocos de código com base no valor da opção selecionada.
  • Console.WriteLine(): Utilizado para exibir o resultado da operação no console.

using System;

class Program
{
    public static void Main(string[] args)
    {
        Console.WriteLine("Selecione uma opção:");
        Console.WriteLine("1 - Adição");
        Console.WriteLine("2 - Subtração");
        Console.WriteLine("3 - Multiplicação");
        Console.WriteLine("4 - Divisão");

        int opcao;
        if (int.TryParse(Console.ReadLine(), out opcao))
        {
            double resultado = 0;

            Console.WriteLine("Digite o primeiro valor:");
            double valor1;
            if (double.TryParse(Console.ReadLine(), out valor1))
            {
                Console.WriteLine("Digite o segundo valor:");
                double valor2;
                if (double.TryParse(Console.ReadLine(), out valor2))
                {
                    switch (opcao)
                    {
                        case 1:
                            resultado = valor1 + valor2;
                            Console.WriteLine("O resultado da adição é: " + resultado);
                            break;
                        case 2:
                            resultado = valor1 - valor2;
                            Console.WriteLine("O resultado da subtração é: " + resultado);
                            break;
                        case 3:
                            resultado = valor1 * valor2;
                            Console.WriteLine("O resultado da multiplicação é: " + resultado);
                            break;
                        case 4:
                            if (valor2 != 0)
                            {
                                resultado = valor1 / valor2;
                                Console.WriteLine("O resultado da divisão é: " + resultado);
                            }
                            else
                            {
                                Console.WriteLine("Não é possível dividir por zero.");
                            }
                            break;
                        default:
                            Console.WriteLine("Opção inválida.");
                            break;
                    }
                }
                else
                {
                    Console.WriteLine("Valor inválido. Certifique-se de digitar um número válido para o segundo valor.");
                }
            }
            else
            {
                Console.WriteLine("Valor inválido. Certifique-se de digitar um número válido para o primeiro valor.");
            }
        }
        else
        {
            Console.WriteLine("Opção inválida. Certifique-se de digitar um número válido para a opção.");
        }
    }
}
